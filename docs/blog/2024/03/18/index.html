Perform analysis without integration

ifnb <- NormalizeData(ifnb) |>
    FindVariableFeatures() |>
    ScaleData() |>
    RunPCA() |>
    FindNeighbors(dims = 1:30, reduction = "pca") |>
    FindClusters(resolution = 2, cluster.name = "unintegrated_clusters") |>
    RunUMAP(
        dims = 1:30, reduction = "pca", reduction.name = "umap.unintegrated"
    )

p1 <- DimPlot(
    ifnb, reduction = "umap.unintegrated",
    group.by = c("stim", "seurat_clusters")
)

Perform integration

ifnb <- IntegrateLayers(
    object = ifnb,
    method = CCAIntegration,
    orig.reduction = "pca",
    new.reduction = "integrated.cca",
    verbose = FALSE
)

### re-join layers after integration
ifnb[["RNA"]] <- JoinLayers(ifnb[["RNA"]])

ifnb <- FindNeighbors(ifnb, reduction = "integrated.cca", dims = 1:30) |>
    FindClusters(resolution = 1) |>
    RunUMAP(dims = 1:30, reduction = "integrated.cca")

p2 <- DimPlot(
    ifnb, reduction = "umap", group.by = c("stim", "seurat_annotations")
)

wrap_plots(p1, p2, ncol = 1)
